#some useful git commands:
git status #to see recent changes
git pull #to update local repository
git reset --hard shaSummOfCommit #to reset all changes and go to changes of mentioned commit
git reset --hard HEAD #to reset all changes and go to changes of last commit
git log #to see information about all commit (SHA-summs, descriptions etc) (to quit press q)

#change the branch
git checkout nameOfBranch
